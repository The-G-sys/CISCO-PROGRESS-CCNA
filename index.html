Switch(config)# int gi0/24

Switch(config-if)# switchport trunk native vlan 99

EtherChannel (LACP)

Switch(config)# int range gi0/1-2
Switch(config-if-range)# channel-group 1 mode active

STP - PortFast

Switch(config-if)# spanning-tree portfast

Router(config)# ip route 0.0.0.0 0.0.0.0 10.0.0.1
Router(config)# ipv6 route 2001:db8::/32 2001:db8::1

OSPFv2 Single-Area

Router(config)# router ospf 1
Router(config-router)# network 192.168.1.0 0.0.0.255 area 0
Router(config-router)# passive-interface gi0/0

HSRP

Router(config-if)# standby 1 ip 192.168.1.1
Router(config-if)# standby 1 priority 110
Router(config-if)# standby 1 preempt

10% 4. IP Services

NAT/PAT

Router(config)# access-list 1 permit 192.168.1.0 0.0.0.255
Router(config)# ip nat inside source list 1 int gi0/0 overload
Router(config-if)# ip nat inside

DHCP Server

Router(config)# ip dhcp excluded-address 192.168.1.1 192.168.1.10
Router(config)# ip dhcp pool LAN
Router(dhcp-config)# network 192.168.1.0 255.255.255.0
Router(dhcp-config)# default-router 192.168.1.1

DHCP Relay

Router(config)# int vlan 10
Router(config-if)# ip helper-address 192.168.1.100

Sécurisation de base

Router(config)# service password-encryption
Router(config)# banner motd #ACCES INTERDIT#

ACL Standard (1-99)

Router(config)# access-list 10 permit 192.168.1.0 0.0.0.255
Router(config)# access-list 10 deny any
Router(config-if)# ip access-group 10 in
